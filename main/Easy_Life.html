Rotate an Array


    public static ArrayList<Integer> rotate(ArrayList<Integer> arr, int n) {
        n = n % arr.size(); // Ensure valid rotation

        ArrayList<Integer> rotated = new ArrayList<>();
        rotated.addAll(arr.subList(arr.size() - n, arr.size())); // Last n elements at start
        rotated.addAll(arr.subList(0, arr.size() - n)); // First part at the end

        return rotated;
    }
